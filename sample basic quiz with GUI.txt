def show_question():
    global current_question
    if current_question >= len(questions):
        result_label.config(text=f"Final Score: {score} out of {len(questions)}")
        return
    q = questions[current_question]
    question_label.config(text=f"Question {current_question + 1}: {q['question']}")
    for i in range(4):
        option_buttons[i].config(text=q['options'][i], state='normal')

def check_answer(i):
    global score
    global current_question
    if questions[current_question]['options'][i] == questions[current_question]['answer']:
        result_label.config(text="Previous Input was Correct", fg='green')
        score += 1
    else:
        result_label.config(text=f"Previous Input was Incorrect\nCorrect Answer: {questions[current_question]['answer']}", fg='red')
    for btn in option_buttons:
        btn.config(state='disabled')
    current_question += 1
    show_question()
